```typescript
export interface FilingRules {
  annual_filing_month?: number    // Calendar month (1-12) of annual filing deadline
  tax_deadline_doy?: number       // Day-of-year for tax return
  agent_renewal_month?: number    // Calendar month of agent renewal
  fiscal_year_end_month?: number  // FY end month (12 for most, 3 for JP/IN)
  fiscal_year_end_day?: number    // FY end day (31 typically)
  grace_period_days?: number      // Buffer days before marking overdue
  [key: string]: unknown
}

export interface Entity {
  id: string
  name: string
  legal_name: string
  entity_type: string
  entity_purpose: EntityPurpose  // 'provider_key' | 'customer_entity' | 'hq'
  jurisdiction_id: string
  incorporation_date: string | null
  status: EntityStatus           // 'active' | 'dormant' | 'dissolving' | 'dissolved'
  registered_agent: RegisteredAgent  // has renewal_date?: string
  // ... other fields
}

export interface Jurisdiction {
  id: string
  country_code: string
  country_name: string
  filing_rules: FilingRules
  // ...
}

export type ComplianceStatus = 'pending' | 'in_progress' | 'completed' | 'overdue'
export type RiskLevel = 'critical' | 'warning' | 'ok'
export type AlertType = 'overdue' | 'due_soon' | 'at_risk' | 'info'
```
